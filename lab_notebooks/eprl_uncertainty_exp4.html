The training phase of this experiment is identical to experiment one.

The testing phase of the experiment has a major change. Before each choice between an object and a circle, participants were shown the circle to be featured in an upcoming choice and were asked to indicate whether that circle was reliable or unreliable (as learned in the training phase).

Most participants still follow the simplified strategy of choosing only circles. A large proportion also follow the optimal choose high old strategy. Neither of these strategies require attending to the reliability of the incremental signal.

However, we also see that now, after forcing participants to attend to the reliability of a circle (as they learned it), many individuals have behavior that is best explained by a strategy that we haven't seen much of before, which is to choose the circle stimulus only if it is reliable. This is interesting because we now have evidence that some participants are making choices considering the incremental stimulus first, rather than the objects, which is necessary for our hypothesis to be tested.

We can see at the group level that this strategy does NOT explain a majority of participants' choices.

Conclusions¶

  • During Training:

    • Participants show a preference for reliable > unreliable circles and high old > low old. This replicates training effects from previous experiments, demonstrating that people can learn the task.

  • During Testing:

    • Participants in general follow the pattern from the previous experiments, where they demonstrate a preference for high old > low old but not reliable > unreliable. However, if we split participants by their accuracy on circle priming, we see that those who correctly indicate circle identity do show the reliable > unreliable choice preference.

    • There is no interaction between value and reliability.

    • The high priming accuracy group chooses old objects > new objects

    • Most participants follow either the choose circle or choosehighold strategy as before. We also see an indication that priming with circle identity influences choices with the emergence of a strategy where some participants choose mostly based on whether a circle is reliable or not.

    • Participants that choose more reliable > unreliable circles also choose more objects when choosing between an object and an unreliable circle, but this is not greater for old objects compared to new objects.

def loadData(dType=None,dataDir="/Users/jonathan/shohamylab/jspych-exps/data/exp4",fileType="stateData"):
    data, exclude_pids = [], [] 
    dataFiles = [d for d in os.listdir(dataDir) if dType in d]
    for f in dataFiles:
        subData = pd.read_csv(os.path.join(dataDir,f))
        if dType == "interaction":
            if len(subData[subData.event == 'blur']) >= exclude_blur:
                exclude_pids.append(f.split('_')[0])
            subData['subjectID'] = [f.split('_')[0]] * len(subData)
        elif dType == "experiment":
            bonus = float(f.split('_')[2].replace('bonus',''))
            subData['bonus'] = [bonus] * len(subData)
        data.append(subData)
    data = pd.concat(data,ignore_index=True)
    return data, exclude_pids

def get_only_acc_trials(pid=None):
    
    inData = data[(data.test_part == "preTest") &
                  (data.subjectID == pid) &
                  (data.trial_number <= 150)]
    preTestAcc = []
    for i_row, row in inData.iterrows():
        if row.deck_type == "orange":
            if row.choice == "Unrel":
                accuracy = 1
            else:
                accuracy = 0
        elif row.deck_type == "blue":
            if row.choice == "Rel":
                accuracy = 1
            else:
                accuracy = 0
        preTestAcc.append(accuracy)
    inData["accuracy"] = preTestAcc
    acc_trials = list(inData[inData.accuracy == 1].trial_number)
    return acc_trials

def get_preTest_accuracy():

    preTestData = data[(data.test_part == "preTest") &
                       (data.trial_number <= 150)]
    preTestAcc = []
    for i_row, row in preTestData.iterrows():
        if row.deck_type == "orange":
            if row.choice == "Unrel":
                accuracy = 1
            else:
                accuracy = 0
        elif row.deck_type == "blue":
            if row.choice == "Rel":
                accuracy = 1
            else:
                accuracy = 0
        preTestAcc.append(accuracy)
    preTestData["accuracy"] = preTestAcc

    preTestAcc = {'pid':[],'accuracy':[]}
    for pid in preTestData.subjectID.unique():
        pidData = preTestData[preTestData.subjectID == pid]
        preTestAcc['pid'].append(pid)

def get_blurs(x):
    if str(x) in n_blurs.subjectID.unique():
        pid_blurs = n_blurs[n_blurs.subjectID == str(x)].n_blurs.iloc[0]
        return pid_blurs
    else:
        # if not there, then they don't have any blurs
        return 0

def get_relative_perf(x):
    # calculate their performance relative to what is expected based on random guessing
    rand_perf = random_perf[random_perf.pid == x].random_test.iloc[0]
    actual_earnings = random_perf[random_perf.pid == x].actual_earnings.iloc[0]
    rel_perf = (actual_earnings*0.01) - (rand_perf*0.01)
    
    return rel_perf

def aggregateTestData(data=None, exclude_pids=None, acc_reliable_only=False):
    
    pids = data.subjectID.unique()

    dOut = {'reliability':[],'value':[],'pOld':[],
            'pid':[],'bonus':[],'propResponse':[],
            'mean_rt':[],'n_rel':[],'n_unrel':[],'n_circle':[],'n_object':[],
            'n_old_high':[],'n_old_low':[],'n_new':[],'rt_old':[],'rt_new':[],
           'rt_rel':[],'rt_unrel':[],'rt_circle':[],'rt_object':[],'n_old_chosen':[],
           'n_new_noprop':[],'n_rel_noprop':[],'n_unrel_noprop':[]}
    for pid in pids:
        
        train_data = data[(data.test_part == 'train') & (data.subjectID == pid)]
        test_data = data[(data.test_part == 'test') &
                         (data.subjectID == pid) &
                         (data.trial_number <= 150)]
        if acc_reliable_only:
            
            acc_trials = get_only_acc_trials(pid=pid)

            try:
                test_data = test_data[test_data.trial_number.isin(acc_trials)]
                if len(test_data) == 0:
                    exclude_pids.append(pid)
            except:
                exclude_pids.append(pid)
                
        bonus = test_data.bonus.mean()

        nTrials = len(train_data)+len(test_data)

        cleaned_train = train_data[train_data.choice != 'no_response']
        cleaned_test = test_data[test_data.choice != 'no_response']
        
        nResponded = len(cleaned_train)+len(cleaned_test)
        proportionResponse = nResponded/nTrials

        old_test = cleaned_test[cleaned_test.old_trial == True]
        old_chosen = old_test[old_test.choice == 'object']

        if proportionResponse >= exclude_response and pid not in exclude_pids:

            colors = ['blue','orange']
            values = ['1.20','0.00']

            for c in colors:
                for v in values:
                    nTotal = len(old_test[(old_test.deck_type == c) & \
                                          (old_test.old_value == v)])
                    nChosen = len(old_chosen[(old_chosen.deck_type == c) & \
                                             (old_chosen.old_value == v)])
                    
                    # number of object trials chosen
                    nOldHigh = len(cleaned_test[(cleaned_test.choice == "object") & \
                                                (cleaned_test.old_trial == True) & \
                                                (cleaned_test.old_value == '1.20')])
                    nOldLow = len(cleaned_test[(cleaned_test.choice == "object") & \
                                               (cleaned_test.old_trial == True) & \
                                               (cleaned_test.old_value == '0.00')])

                    nNew = len(cleaned_test[(cleaned_test.choice == "object") & \
                                            (cleaned_test.old_trial == False) & \
                                            (cleaned_test.deck_type == c)])
                    nObject = nOldHigh+nOldLow+nNew
                    
                    rtOld = cleaned_test[(cleaned_test.choice == "object") & \
                                            (cleaned_test.old_trial == True)].rt.mean()
                    rtNew = cleaned_test[(cleaned_test.choice == "object") & \
                                            (cleaned_test.old_trial == False)].rt.mean()
                    rtObject = rtOld+rtNew
                    
                    # number of circle trials chosen
                    nReliable = len(cleaned_test[(cleaned_test.choice == "blue")])
                    nUnreliable = len(cleaned_test[(cleaned_test.choice == "orange")])
                    nCircle = nReliable+nUnreliable
            
                    rtReliable = cleaned_test[cleaned_test.choice == "blue"].rt.mean()
                    rtUnreliable = cleaned_test[cleaned_test.choice == "orange"].rt.mean()
                    rtCircle = np.mean([rtReliable,rtUnreliable])
                    
                    nTotalChoices = nObject+nCircle
                    nNewNoProp = nNew
                    nRelNoProp = nReliable
                    nUnrelNoProp = nUnreliable
                    # set in terms of proportions
                    nObject = nObject/nTotalChoices                        
                    nNew = nNew/nTotalChoices
                    nOldHigh = nOldHigh/nTotalChoices
                    nOldLow = nOldLow/nTotalChoices
                    nCircle = nCircle/nTotalChoices
                    nReliable = nReliable/nTotalChoices
                    nUnreliable = nUnreliable/nTotalChoices
                                        
                    # number of old chosen/number of total old trials
                    try:
                        pOld = nChosen/nTotal
                    except:
                        pOld = np.nan
                    nOldChosen = nChosen
                    
                    # mean RT for when the old items are chosen
                    meanRT = old_test[(old_test.deck_type == c) & \
                                    (old_test.old_value == v)].rt.mean()
                    
                    if c == 'blue':
                        rel = 'reliable'
                    else:
                        rel = 'unreliable'

                    if v == '1.20':
                        val = 'high'
                    else:
                        val = 'low'

                    dOut['reliability'].append(rel)
                    dOut['value'].append(val)
                    dOut['pOld'].append(pOld)
                    dOut['pid'].append(pid)
                    dOut['bonus'].append(bonus)
                    dOut['propResponse'].append(proportionResponse)
                    dOut['mean_rt'].append(meanRT)
                    dOut['n_rel'].append(nReliable)
                    dOut['n_unrel'].append(nUnreliable)
                    dOut['n_circle'].append(nCircle)
                    dOut['n_old_high'].append(nOldHigh)
                    dOut['n_old_low'].append(nOldLow)
                    dOut['n_new'].append(nNew)
                    dOut['n_new_noprop'].append(nNewNoProp)
                    dOut['n_rel_noprop'].append(nRelNoProp)
                    dOut['n_unrel_noprop'].append(nUnrelNoProp)
                    dOut['n_object'].append(nObject)
                    dOut['n_old_chosen'].append(nOldChosen)
                    dOut['rt_rel'].append(rtReliable)
                    dOut['rt_unrel'].append(rtUnreliable)
                    dOut['rt_circle'].append(rtCircle)
                    dOut['rt_old'].append(rtOld)
                    dOut['rt_new'].append(rtNew)
                    dOut['rt_object'].append(rtObject)


    dOut = pd.DataFrame.from_dict(dOut)
    return dOut

def gen_random_performance(data=None,n_sims=None):

    # Let's try and model guessing behavior
    # e.g. how much would we expect them to earn if they were just guessing?
    import random

    test_data = data[(data.test_part == 'test')]

    random_perf = {'pid':[],'random_test':[],'actual_earnings':[]}
    for pid in test_data.subjectID.unique():

        pid_data = test_data[test_data.subjectID == pid]

        earnings = pid_data[pid_data.earned != 'none'].earned
        total_earnings = np.sum(pd.to_numeric(earnings))

        sim_earnings = []
        for sim in range(n_sims):

            test_earnings = 0

            for i_row, row in pid_data.iterrows():

                reliableSample = np.round_(np.random.normal(0.6,0.05,1),decimals=2)
                unreliableSample = np.round_(np.random.normal(0.6,0.2,1),decimals=2)

                old_val = row.old_value
                deck_type = row.deck_type

                if old_val == 'none':
                    obj_val = random.choice([0,1.2])
                elif old_val == '0.00':
                    obj_val = 0
                elif old_val == '1.20':
                    obj_val = 1.2

                if deck_type == 'blue':
                    deck_val = reliableSample[0]
                else:
                    deck_val = unreliableSample[0]

                test_earnings+=random.choice([deck_val,obj_val])

            sim_earnings.append(test_earnings)

        random_perf['pid'].append(pid)
        random_perf['random_test'].append(np.mean(sim_earnings))
        random_perf['actual_earnings'].append(total_earnings)

    random_perf = pd.DataFrame.from_dict(random_perf)

    return random_perf

def gen_strategy(data=None,strategy=None,get_only_acc=False):
    to_exclude = []

    # Model object-centric choice strategy
    # Choose an object only if it is high-value old
    import random

    test_data = data[(data.test_part == 'test')]

    perf = {'pid':[],strategy:[]}
    for pid in test_data.subjectID.unique():

        pid_data = test_data[test_data.subjectID == pid]
        pid_data = pid_data[pid_data.trial_number <= 150]
        
        if get_only_acc:
        
            acc_trials = get_only_acc_trials(pid=pid)

            try:
                pid_data = pid_data[pid_data.trial_number.isin(acc_trials)]
                if len(pid_data) == 0:
                    to_exclude.append(pid)
            except:
                to_exclude.append(pid)

        if pid not in to_exclude:
            earnings = pid_data[pid_data.earned != 'none'].earned
            total_earnings = np.sum(pd.to_numeric(earnings))

            agreement = 0
            for i_row, row in pid_data.iterrows():

                old_val = row.old_value
                deck_type = row.deck_type

                if strategy == 'choosehighold':

                    if old_val == '1.20':
                        sim_choice = 'object'
                    else:
                        sim_choice = 'deck'

                elif strategy == 'choosehighandnew':
                    if old_val == '1.20' or old_val == 'none':
                        sim_choice = 'object'
                    else:
                        sim_choice = 'deck'

                elif strategy == 'chooseold':

                    if old_val == '1.20' or old_val == '0.00':
                        sim_choice = 'object'
                    else:
                        sim_choice = 'deck'

                elif strategy == 'choosecircle':

                    sim_choice = 'deck'

                elif strategy == 'chooseobject':

                    sim_choice = 'object'

                elif strategy == 'chooserel':

                    if deck_type == 'blue':
                        sim_choice = 'deck'
                    else:
                        sim_choice = 'object'

                elif strategy == 'chooseunrel':

                    if deck_type == 'orange':
                        sim_choice = 'deck'
                    else:
                        sim_choice = 'object'

                elif strategy == 'random':

                    sim_choice = random.choice(['object','deck'])

                if sim_choice == 'object':
                    if row.choice =='object':
                        agree = 1
                    else:
                        agree = 0
                elif sim_choice == 'deck':
                    if row.choice == 'blue' or row.choice == 'orange':
                        agree = 1
                    else:
                        agree = 0

                agreement+=agree

            agreement = agreement/len(pid_data)

            perf['pid'].append(pid)
            perf[strategy].append(agreement)

    perf = pd.DataFrame.from_dict(perf)
    
    return perf

def get_best_strategy(strategyDf):
    global s
    s = {'pid':[],'bestStrategy':[],'value':[]}
    for pid in strategyDf.pid.unique():
        pidData = strategyDf[strategyDf.pid == pid]
        maxRow = pidData.loc[pidData.value.argmax()]
        bestStrategy = maxRow.variable
        maxValue = maxRow.value
        s['pid'].append(pid)
        s['bestStrategy'].append(bestStrategy)
        s['value'].append(maxValue)
    s = pd.DataFrame.from_dict(s)
    s1 = s.groupby('bestStrategy').size().reset_index(name='tally')
    return s1

def perform_median_split():
    
    preTestAcc = get_preTest_accuracy()

    preTestAccToAdd = []
    for pid in agg_data.pid.unique():
        acc = preTestAcc[preTestAcc.pid == pid].accuracy.iloc[0]
        pidDf = agg_data[agg_data.pid == pid]
        preTestAccToAdd.append([acc] * len(pidDf))
    preTestAccToAdd = [i for j in preTestAccToAdd for i in j]
    agg_data['preTest_accuracy'] = preTestAccToAdd

    median = agg_data.preTest_accuracy.median()
    print("PreTest median performance is %s"%(median))
    
    median_split = {'pid':[],'group':[]}
    for pid in agg_data.pid.unique():
        pid_accuracy = agg_data[(agg_data.pid == pid)] \
                                .preTest_accuracy.iloc[0]
        if pid_accuracy > median:
            median_split['group'].append('high')
        else:
            median_split['group'].append('low')
        median_split['pid'].append(pid)
    median_split = pd.DataFrame.from_dict(median_split)

    group_assignments = []
    for pid in agg_data.pid.unique():
        group_assignment = median_split[median_split.pid == pid].group.iloc[0]
        group_assignments.append([group_assignment] * len(agg_data[agg_data.pid == pid]))
    group_assignments = [i for j in group_assignments for i in j]
    agg_data['median_split'] = group_assignments
    
    for g in ["low","high"]:
        n_pids = len(agg_data[agg_data['median_split'] == g].pid.unique())
        print("N participants in %s group: %s"%(g, n_pids))
    
    return agg_data

def get_senstivity_relationship():
    pidList = list(np.array(agg_data[agg_data.reliability == "reliable"].pid))
    
    def get_choice_diff(value=None,choice_type=None):
        rel = np.array(agg_data[(agg_data.reliability == "reliable") & \
                                (agg_data.value == value)][choice_type])
        unrel = np.array(agg_data[(agg_data.reliability == "unreliable") & \
                                (agg_data.value == value)][choice_type])
        return rel - unrel
        
    highOldChosenDiff = get_choice_diff(value="high",choice_type="n_old_chosen")
    lowOldChosenDiff = get_choice_diff(value="low",choice_type="n_old_chosen")
    newChosenDiff = get_choice_diff(value="high",choice_type="n_new_noprop")

    reliabilityDiff = np.array(agg_data[agg_data.reliability == "reliable"].n_rel_noprop) - \
                      np.array(agg_data[agg_data.reliability == "unreliable"].n_unrel_noprop)
    reliabilityDiff = reliabilityDiff[:len(reliabilityDiff):2]
    pidList = pidList[:len(pidList):2]
    
    chosenDiff = [i for j in [lowOldChosenDiff,highOldChosenDiff,newChosenDiff] for i in j]
    pidList = [i for j in [pidList,pidList,pidList] for i in j]
    reliabilityDiff = [i for j in [reliabilityDiff, reliabilityDiff,reliabilityDiff] for i in j]
    value = [i for j in [["low"] * len(lowOldChosenDiff),
                         ["high"] * len(highOldChosenDiff),
                         ["new"] * len(newChosenDiff)] for i in j]

    relDiffDf = {'pid':pidList,
                 'rel_chosen - unrel_chosen':chosenDiff,
                 'n_rel - n_unrel':reliabilityDiff,
                 'value':value}

    relDiffDf = pd.DataFrame.from_dict(relDiffDf)
    
    return relDiffDf
